# 딕셔너리


```python
>>> 딕셔너리 = {
...     "키A": "값A",
...     "키b": "값b",
...     "키c": "값c",
...     "키d": "값d",
...     "문자열": "값",
...     273 : [1,2,3,4],
...     True: False
... }
>>>
>>> print(딕셔너리["문자열"])
값
>>> print(딕셔너리[273])
[1, 2, 3, 4]
>>> print(딕셔너리[True])
False
>>>
```

```python
>>> for key in 딕셔너리:
...     print("{} : {}".format(key, 딕셔너리[key]))
...
키A : 값A
키b : 값b
키c : 값c
키d : 값d
문자열 : 값값
273 : [1, 2, 3, 4]
True : False
>>>
```
